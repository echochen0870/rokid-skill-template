# rokid-skill-template
# 用于快速创建媒体流歌单技能的模板
## 支持情况
> 指令（可在media_NLP.json内自行扩充)
> - 开始播放、
> - 下一首、下一曲、下一个、切歌
> - 上一首、上一曲、上一个
> - 暂停、暂停播放、暂停一下、
> - 继续、继续播放

> 服务脚本编辑
> - //播放模式，0：顺序播放，1：随机播放。
var play_mode = 0;
> - //播放音频的索引。
var play_count = 0;
> - //tts欢迎词
var WELCOME_TTS = '欢迎来到通用媒体模板';
> - //播放列表，顺序播放会按照列表顺序播放。
var play_list = [];
